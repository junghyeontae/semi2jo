
--사원
CREATE TABLE EMPLOYEE(
  EMP_CODE VARCHAR2(3),
  PWD VARCHAR2(500),
  EMP_NAME VARCHAR2(20) NOT NULL,
  PERSONAL_ID CHAR(14) NOT NULL,
  EMAIL VARCHAR2(25),
  GENDER CHAR(1 BYTE) CHECK(GENDER IN('M','F')),
  AGE NUMBER(2),
  SALARY NUMBER(10),
  PHONE VARCHAR2(13),
  ADDRESS VARCHAR2(500) NOT NULL,
  DEPT_CODE CHAR(2),
  JOB_CODE CHAR(2),
  MANAGER_YN VARCHAR2(1) CHECK(MANAGER_YN IN('Y','N')),
  HIRE_DATE DATE,
  ETC VARCHAR2(300),
  SUB_DEPT CHAR(2)

);

  CREATE TABLE BOARD_NORMAL(
  BNO NUMBER NOT NULL,
  BTYPE NUMBER,

  BWRITER VARCHAR2(30),
  BCOUNT NUMBER,

  );
